[254. Factor Combinations](https://leetcode.com/problems/factor-combinations/)  

## recursive pass
```
class Solution {
    
    recursive
    func getFactors(_ n: Int) -> [[Int]] {
        var res = [[Int]]()
        func dfs(_ curr: Int, _ path: [Int], _ start: Int) {
            var n = start
            while n*n <= curr {
                if curr % n == 0 { 
                    res.append(path+[n, curr/n])
                    dfs(curr/n, path + [n], n)
                }
                n += 1
            }
        }
        dfs(n, [], 2)
        return res
    }
```


## iterative, fastest
```
func getFactors(_ n: Int) -> [[Int]] {
        var res = [[Int]]()
        var queue = [(n, [Int](), 2)]
        while !queue.isEmpty {
            let (curr, path, start) = queue.removeFirst()
            var n = start
            while n*n <= curr {
                if curr % n == 0 {
                    res.append(path+[n, curr/n])
                    queue.append((curr/n,path+[n],n))
                }
                n += 1
            }
        }
        return res
    }
}
```

## Time limit exceed
```
class Solution {
    var ans = [[Int]]()
    func getFactors(_ n: Int) -> [[Int]] {
        helper(n, 2)
        return ans
    }
    
    func helper(_ n: Int, _ start: Int, _ path: [Int] = []) {
        if n <= 1 {
            if path.count > 1 {
                ans.append(path)
            }
        } else if start <= n {
            for i in start...n {
                if n % i == 0 {
                    helper(n / i, i, path + [i])
                }
            }
        }
    }
}
```

## Time limit excced
```
class Solution {
    var ans = [[Int]]()
    func getFactors(_ n: Int) -> [[Int]] {
        var path = [Int]()
        helper(n, 2, &path)
        return ans
    }
    
    func helper(_ n: Int, _ start: Int, _ path: inout [Int]) {
        guard start * start <= n else {return}
        for i in start...n where n % i == 0 {
            if i * i > n {break}
            path.append(i)
            path.append(n / i)
            ans.append(path)
            path.removeLast()
            helper(n / i, i, &path)
            path.removeLast()
        }
    }
}
```
