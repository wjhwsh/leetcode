#[251. Flatten 2D Vector](https://leetcode.com/problems/flatten-2d-vector/)
swift flatMap can flat array, such as [[2, 3], 4, [5, 6]].flatMap {$0} become [2, 3, 4, 5, 6]

```
class Vector2D {
    var data2D = [Int]()
    var index = 0
    init(_ vec: [[Int]]) {
        data2D = vec.flatMap {$0}
    }
    
    func next() -> Int {
        defer {if index < data2D.count {index += 1}}
        return data2D[index]
    }
    
    func hasNext() -> Bool {
        index < data2D.count
    }
}

```
