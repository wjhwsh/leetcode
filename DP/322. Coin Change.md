[322. Coin Change](https://leetcode.com/problems/coin-change/)  
```
class Solution {
    func coinChange(_ coins: [Int], _ amount: Int) -> Int {
        var dp = Array(repeating: Int.max - 1, count: amount+1)
        dp[0] = 0
        for coin in coins {
            var i = coin
            while i <= amount {
                dp[i] = min(dp[i], dp[i-coin] + 1)
                i += 1
            }
        }
        if dp[amount] < amount + 1 {
            return dp[amount]
        } else {
            return -1
        }
    }
}
```
