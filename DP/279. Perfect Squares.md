[279. Perfect Squares](https://leetcode.com/problems/perfect-squares/)  
```
class Solution {
    func numSquares(_ n: Int) -> Int {
        var dp = Array(repeating: Int.max, count: n+1)
        dp[0] = 0
        dp[1] = 1
        if n < 2 {return dp[n]}
        for i in 2...n {
            var j = 1
            while j * j <= i {
                dp[i] = min(dp[i], dp[i-j*j] + 1)
                j += 1
            }
        }
        return dp[n]
    }
}
```
