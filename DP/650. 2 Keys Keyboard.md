[650. 2 Keys Keyboard](https://leetcode.com/problems/2-keys-keyboard/)  

```
5ms
class Solution {
    func minSteps(_ n: Int) -> Int {
        var ans = n
        if n == 1 {return 0}
        else if n == 2 {return 2}
        else if n == 3 {return 3}
        for i in 1..<n {
            if n % i == 0 {
                ans = min(ans, minSteps(i) + (n / i))
            }
        }
        return ans
    }
}
```

```
class Solution {
    func minSteps(_ n: Int) -> Int {
        var dp = Array(repeating: n, count: n + 1)
        dp[1] = 0
        for i in stride(from: 2, through: n, by: 1) {
            for j in 1..<i {
                if i % j == 0 {
                    dp[i] = min(dp[i], dp[j] + i / j)
                }
            }
        }
        return dp[n]
    }
}
```

```
class Solution {
    func minSteps(_ n: Int) -> Int {
        var ans = 0
        var n = n
        var d = 2
        while n > 1 {
            while n % d == 0 {
                ans += d
                n = n / d
            }
            d += 1
        }
        return ans
    }
}
```
