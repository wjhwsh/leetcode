[983. Minimum Cost For Tickets](https://leetcode.com/problems/minimum-cost-for-tickets/)  

```
class Solution {
    func mincostTickets(_ days: [Int], _ costs: [Int]) -> Int {
        let n = days.count
        let duration = [1, 7, 30]
        var memo = Array(repeating: -1, count: n)
        func dp(_ index: Int) -> Int {
            if index >= n {
                return 0
            } 
            if memo[index] != -1 {return memo[index]}
            var ans = Int.max
            var k = index
            for j in 0..<3 {
                while k < days.count && days[k] - days[index] < duration[j] {
                    k += 1
                }
                ans = min(ans, costs[j] + dp(k))
            }
            memo[index] = ans
            return ans
        }
        
        return dp(0)
    }
        
}
```
